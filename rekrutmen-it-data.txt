SELECT nama AS cabkor
FROM `rekrutmenitdata.data_cabang`
WHERE
  nama AS cabkor = 'Kantor Cabang Koordinator%'
),
cabang AS(
SELECT nama AS cabang

),

FROM `rekrutmenitdata.data_cabang`

  cabang = '%Cabang%'
)

  cabkor, 
  cabang,
  unit_kerja
FROM cabkor, cabang, unit_kerja
